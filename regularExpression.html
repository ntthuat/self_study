Quy tac ve so luong ky tu

chuoi goc: {{chuoi4}}
"x?" nghia la mot ky tu x hoac khong co ky tu nao: chuoi4a = chuoi4.replace(/go?d/gi, '*')
chuoi 4a: {{chuoi4a}}
"x*" nghia la mot hoac nhieu ky tu x hoac khong co ky tu nao
"x+" nghia la co it nhat mot ky tu x
"x{n}" nghia la ky tu x lap lai n lan
"x{n,}" nghia la ky tu x lap lai it nhat n lan
"x{n,m}" nghia la ky tu x lap lai tu n toi m lan
var str = 'good night, godgad, gd, gooogle';
var str6 = str.replace(/go?d/gi, '*');
// str6: good night, *gad, *, gooogle
var str7 = str.replace(/g(oo)?d/gi, '*');
// str7: * night, godgad, *, gooogle
var str8 = str.replace(/go*d/gi, '*');
// str8: * night, *gad, *, gooogle
var str9 = str.replace(/go+/gi, '*');
// str9: *d night, *dgad, gd, *gle
var str10 = str.replace(/o{2}/gi, '*');
// str10: g*d night, godgad, gd, gooogle
var str11 = str.replace(/o{2,}/gi, '*');
// str11: g*d night, godgad, gd, g*gle
var str12 = str.replace(/go{1,3}d/gi, '*');
// str12: *d night, *dgad, gd, gooogle

Quy tac ve danh sach ky tu

"[...]" trong dau ngoac vuong liet ke danh sach nhung ky tu hop quy tac, co the dung dau "-" giua 2 ky tu de the hien danh sach nhung ky tu trong khoang 2 ky tu theo vi tri ASCII
"[^...]" nguoc lai voi "[...]", thay vi liet ke nhung ky tu hop quy tac, no cho biet ky tu khong hop quy tac
vi du:
var str ='ABCDEF, abcdef, 12345';
var str13 = str.replace(/[Ab1]/g, '*');
// str13: *BCDEF, a*cdef, *2345 var str14 = str.replace(/[A-Cd-f1-35]/g, '*');
// str14: ***DEF, abc***, ***4*
var str15 = str.replace(/[^A-f4-5]/g, '*');
// str15: ABCDEF**abcdef*****45
